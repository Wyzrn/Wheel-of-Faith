<objective>
Implement score-based redemption probability: the chance of landing "Redemption" on the
redemptionSpin wheel is determined by the character's overall score, not a fixed 25%.

A low-scoring character (overall score near 0) should have up to an 85% chance.
A high-scoring character (near 100) should have as low as 5%.
The midpoint (score 50) lands around 25%.

Formula: p = max(0.05, min(0.85, 0.05 + 0.80 * (1 - score/100)^2))

This wires the Redemption Spin to gameplay — playing well costs you the redemption safety net.
</objective>

      const p = 0.05 + 0.80 * Math.pow(1 - overallScore / 100, 2)

    ```typescript
    import { redemptionProbability } from '../src/lib/game/redemption.js'

    const TRIALS = 10_000

    // Simulate random overall scores drawn from a roughly realistic distribution.
    // Mix of uniform random + some bias toward middle tiers.
    function randomScore(): number {
      // Average of 3 uniform [0,100] draws — produces bell-ish distribution centered ~50
      return (Math.random() * 100 + Math.random() * 100 + Math.random() * 100) / 3
    }

    let redemptionCount = 0
    const buckets: Record<string, number> = { '0-20': 0, '21-40': 0, '41-60': 0, '61-80': 0, '81-100': 0 }

    for (let i = 0; i < TRIALS; i++) {
      const score = randomScore()
      const p = redemptionProbability(score)
      if (Math.random() < p) redemptionCount++

      if (score <= 20) buckets['0-20']++
      else if (score <= 40) buckets['21-40']++
      else if (score <= 60) buckets['41-60']++
      else if (score <= 80) buckets['61-80']++
      else buckets['81-100']++
    }

    const triggerRate = (redemptionCount / TRIALS * 100).toFixed(2)
    console.log(`\nRedemption Simulation — ${TRIALS.toLocaleString()} trials`)
    console.log(`Trigger rate: ${triggerRate}% (target: 25–35%)`)
    console.log(`Triggered: ${redemptionCount}`)
    console.log(`\nScore distribution:`)
    for (const [range, count] of Object.entries(buckets)) {
      console.log(`  ${range}: ${count} (${(count / TRIALS * 100).toFixed(1)}%)`)
    }

    // Spot-check key probabilities
    console.log(`\nProbability spot-check:`)
    for (const score of [0, 25, 50, 75, 100]) {
      console.log(`  Score ${score}: ${(redemptionProbability(score) * 100).toFixed(1)}%`)
    }
    ```
